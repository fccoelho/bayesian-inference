 1  """ 
 2  Example of an SEIR model with two Infectious classes: subclinical(Is) and clinical(Ic) 
 3          Is 
 4         /  \ 
 5  S -> E     R 
 6         \  / 
 7          Ic 
 8           
 9  States: 
10  S: Susceptible 
11  E: Exposed 
12  Is: Infectious subclinical 
13  Ic: Infectious clinical 
14  R: Recovered 
15   
16  Transition rates: 
17  b,ks,kc,rs,rc = (0.001, 0.1, 0.1, 0.01, .01) 
18  Transitions: 
19  S -> E : b*S*(Is+Ic) 
20  E -> Is : ks*E 
21  E -> Ic : kc*E 
22  Is -> R : rs*Is 
23  Ic -> R : rc*Ic 
24   
25  """ 
26  from gillespie import Model 
27  import time 
28  from numpy import array 
29  vars = ['S','E','Is','Ic','R'] 
30  #rates: b,ks,kc,rs,rc 
31   
32  r = (0.001, 0.1, 0.1, 0.01, .01) 
33  ini = (490,0,10,0,0) 
34  prop = (lambda r,ini:r[0]*ini[0]*(ini[2]+ini[3]), 
35          lambda r,ini:r[1]*ini[1], 
36          lambda r,ini:r[2]*ini[1], 
37          lambda r,ini:r[3]*ini[2], 
38          lambda r,ini:r[4]*ini[3] 
39          ) 
40   
41  tmat = array([[-1,0,0,0,0], 
42              [1,-1,-1,0,0], 
43              [0,1,0,-1,0], 
44              [0,0,1,0,-1], 
45              [0,0,0,1,1] 
46              ]) 
47  #for e in prop: 
48  #    print e() 
49  M=Model(vnames=vars,rates = r,inits=ini,tmat=tmat,propensity=prop) 
50  t0 = time.time() 
51  M.run(tmax=80,reps=100) 
52  print 'total time: ',time.time()-t0 
53  t,series,steps = M.getStats() 
54  print steps,'steps' 
55  from pylab import plot , show, legend, errorbar 
56  plot(t,series.mean(axis=2),'-o') 
57  legend(vars,loc=0) 
58  show()
